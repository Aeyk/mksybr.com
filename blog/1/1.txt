#+BEGIN_SRC lisp

  ;; RGB now
  (defvar rp 3)
  (defvar gp 5)
  (defvar bp 6)

  (defvar dly 2) 

 
  (defun stp ()
    (pinmode rp t)
    (pinmode gp t)
    (pinmode bp t)) 


  (defun red (n)
    (analogwrite rp n))

  (defun gre (n)
    (analogwrite gp n))

  (defun blu (n)
    (analogwrite bp n))

  (defun rgb (a b c)
    (red a)
    (gre b)
    (blu c))

  (defun off ()
    (red 255)
    (gre 255)
    (blu 255))


  (defun on ()
    (red 0)
    (gre 0)
    (blu 0))

  (defun ron ()
    (red (random 255))
    (gre (random 255))
    (blu (random 255)))


  (defun rxn ()
    (red (logxor 255 (random 255)))
    (gre (logxor 255 (random 255)))
    (blu (logxor 255 (random 255))))


  (defvar max 256)
  (defvar dll 500)

  (dotimes (x 4)
    (rxn)
    (delay 800)
    (ron)
    (delay 800)
    (off))
#+END_SRC
